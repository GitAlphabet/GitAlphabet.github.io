虚拟滚动 ​

虚拟滚动（Virtual Scrolling）是一种前端优化技术，用于高效渲染大量数据列表。当需要渲染成千上万个列表项时，传统做法会导致性能问题，而虚拟滚动只渲染可视区域内的元素，从而大幅提升性能。

为什么需要虚拟滚动？ ​

当处理大量数据时，传统滚动方式存在以下问题：

  1. DOM 节点过多：渲染成千上万个 DOM 元素会导致内存占用过高
  2. 渲染性能下降：浏览器需要处理大量元素的重绘和重排
  3. 滚动卡顿：大量元素导致滚动不流畅
  4. 内存浪费：不可见的元素仍然占用资源

虚拟滚动通过只渲染可视区域的元素，解决了这些问题。

核心原理 ​

虚拟滚动的核心思想是：

  1. 固定容器高度：设置一个固定高度的滚动容器
  2. 计算可视区域：根据容器高度和滚动位置计算当前可见的元素范围
  3. 动态渲染：只渲染当前可见的元素，其余元素用占位符替代
  4. 更新位置：根据滚动位置动态更新元素位置

2. 计算可见范围 ​

javascript
/**
 * 计算当前可视区域的元素索引范围
 * @param {HTMLElement} container - 滚动容器元素
 * @param {number} itemHeight - 每个列表项的高度
 * @returns {Object} - 包含起始索引和结束索引的对象
 */
function getVisibleRange(container, itemHeight) {
  // 获取当前滚动位置（距离顶部的距离）
  const scrollTop = container.scrollTop;
  // 获取容器可见区域的高度
  const containerHeight = container.clientHeight;
  
  // 计算第一个可见元素的索引：用滚动位置除以每个元素的高度，向下取整
  const startIndex = Math.floor(scrollTop / itemHeight);
  
  // 计算最后一个可见元素的索引：用滚动位置+容器高度除以每个元素的高度，向上取整
  const endIndex = Math.ceil((scrollTop + containerHeight) / itemHeight);
  
  return { startIndex, endIndex };
}

3. 完整实现示例 ​

javascript
/**
 * 虚拟滚动类
 * 实现只渲染可视区域内元素的滚动效果
 */
class VirtualScroll {
  /**
   * 构造函数
   * @param {Object} options - 配置选项
   * @param {HTMLElement} options.container - 滚动容器元素
   * @param {number} options.itemHeight - 每个列表项的固定高度
   * @param {number} options.totalItems - 列表项总数
   * @param {Function} options.renderItem - 渲染列表项的函数
   */
  constructor(options) {
    // 保存配置参数
    this.container = options.container;
    this.itemHeight = options.itemHeight;
    this.totalItems = options.totalItems;
    this.renderItem = options.renderItem;
    
    // 创建滚动内容容器，用于放置实际渲染的元素
    this.scrollContent = document.createElement('div');
    this.container.appendChild(this.scrollContent);
    
    // 存储当前可见的DOM元素列表
    this.visibleItems = [];
    // 缓存已经创建过的元素，避免重复创建
    this.itemCache = new Map();
    
    // 初始化虚拟滚动
    this.init();
  }
  
  /**
   * 初始化虚拟滚动
   * 设置内容容器高度，添加滚动事件监听
   */
  init() {
    // 设置滚动内容的总高度（所有元素高度之和）
    this.scrollContent.style.height = `${this.totalItems * this.itemHeight}px`;
    
    // 初始渲染可见区域
    this.update();
    
    // 监听滚动事件，滚动时更新可见区域
    this.container.addEventListener('scroll', () => {
      this.update();
    });
  }
  
  /**
   * 更新可视区域
   * 计算当前可见的元素范围，并重新渲染
   */
  update() {
    // 获取当前可见的索引范围
    const { startIndex, endIndex } = this.getVisibleRange();
    // 创建文档片段，用于批量添加元素，减少DOM操作
    const fragment = document.createDocumentFragment();
    
    // 清除当前可见的所有元素
    this.visibleItems.forEach(item => {
      item.remove();
    });
    this.visibleItems = [];
    
    // 渲染新的可见元素
    for (let i = startIndex; i <= endIndex; i++) {
      // 从缓存中获取元素，避免重复创建
      let item = this.itemCache.get(i);
      
      // 如果缓存中没有，则创建新元素
      if (!item) {
        item = this.renderItem(i);
        this.itemCache.set(i, item);
      }
      
      // 设置元素的绝对定位
      item.style.position = 'absolute';
      // 根据索引计算元素在垂直方向的位置
      item.style.top = `${i * this.itemHeight}px`;
      // 设置元素宽度占满容器
      item.style.width = '100%';
      
      // 添加到文档片段和可见元素列表
      fragment.appendChild(item);
      this.visibleItems.push(item);
    }
    
    // 清空滚动内容容器并添加新的可见元素
    this.scrollContent.innerHTML = '';
    this.scrollContent.appendChild(fragment);
  }
  
  /**
   * 获取当前可视区域的索引范围
   * 考虑边界情况，确保索引不超出范围
   * @returns {Object} - 包含起始索引和结束索引的对象
   */
  getVisibleRange() {
    const scrollTop = this.container.scrollTop;
    const containerHeight = this.container.clientHeight;
    
    // 计算起始索引，确保不小于0
    const startIndex = Math.max(0, Math.floor(scrollTop / this.itemHeight));
    // 计算结束索引，确保不超过总项数-1
    const endIndex = Math.min(
      this.totalItems - 1,
      Math.ceil((scrollTop + containerHeight) / this.itemHeight)
    );
    
    return { startIndex, endIndex };
  }
}

自定义 React 虚拟滚动 ​

jsx
import React, { useState, useEffect, useRef } from 'react';

/**
 * 自定义虚拟滚动列表组件
 * @param {Object} props - 组件属性
 * @param {Array} props.items - 列表项数据
 * @param {number} props.itemHeight - 每个列表项的高度
 * @param {number} props.height - 容器高度
 * @returns {JSX.Element} - 虚拟滚动列表
 */
const VirtualScrollList = ({ items, itemHeight, height }) => {
  // 起始索引，记录当前可视区域的第一个元素索引
  const [startIndex, setStartIndex] = useState(0);
  // 结束索引，使用ref避免在渲染中频繁更新
  const endIndexRef = useRef(0);
  // 容器元素的引用
  const containerRef = useRef(null);
  
  // 使用useEffect监听滚动事件，更新可见区域
  useEffect(() => {
    const container = containerRef.current;
    // 更新可见范围的函数
    const updateVisibleRange = () => {
      // 获取滚动位置
      const scrollTop = container.scrollTop;
      // 获取容器可见区域高度
      const containerHeight = container.clientHeight;
      
      // 计算新的起始索引
      const newStartIndex = Math.floor(scrollTop / itemHeight);
      // 计算新的结束索引
      const newEndIndex = Math.ceil((scrollTop + containerHeight) / itemHeight);
      
      // 更新状态
      setStartIndex(newStartIndex);
      endIndexRef.current = newEndIndex;
    };
    
    // 添加滚动事件监听器
    container.addEventListener('scroll', updateVisibleRange);
    // 初始调用一次，设置初始可见范围
    updateVisibleRange();
    
    // 组件卸载时移除事件监听器
    return () => container.removeEventListener('scroll', updateVisibleRange);
  }, [itemHeight]);
  
  // 获取当前可见的列表项
  const visibleItems = items.slice(
    startIndex,           // 起始索引
    endIndexRef.current + 1 // 结束索引+1（slice是左闭右开）
  );
  
  // 计算总高度（所有项的高度之和）
  const totalHeight = items.length * itemHeight;
  // 计算偏移量（起始元素距离顶部的距离）
  const offsetY = startIndex * itemHeight;
  
  return (
    <div
      ref={containerRef}
      style={{ height, overflowY: 'auto' }}
    >
      {/* 外层容器，设置总高度 */}
      <div style={{ height: totalHeight, position: 'relative' }}>
        {/* 内层容器，用于定位可见的元素 */}
        <div style={{ position: 'absolute', top: 0, left: 0, right: 0 }}>
          {/* 遍历渲染可见的元素 */}
          {visibleItems.map((item, index) => (
            <div
              key={startIndex + index}  // 使用起始索引+当前索引作为key，确保唯一性
              style={{
                position: 'absolute',
                // 计算元素在垂直方向的位置
                top: `${(startIndex + index) * itemHeight}px`,
                height: itemHeight,      // 设置固定高度
                width: '100%'            // 占满宽度
              }}
            >
              {item}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

性能优化技巧 ​

1. 节流滚动事件 ​

javascript
/**
 * 节流函数，限制函数在一定时间内只能执行一次
 * @param {Function} fn - 要节流的函数
 * @param {number} delay - 延迟时间（毫秒）
 * @returns {Function} - 节流后的函数
 */
function throttle(fn, delay) {
  let lastCall = 0;
  return function (...args) {
    const now = new Date().getTime();
    if (now - lastCall >= delay) {
      lastCall = now;
      return fn.apply(this, args);
    }
  };
}

// 使用节流优化滚动事件处理，限制为60fps（约16ms一次）
const throttledUpdate = throttle(this.update.bind(this), 16);
this.container.addEventListener('scroll', throttledUpdate);

2. 使用 Intersection Observer ​

javascript
/**
 * 使用 Intersection Observer API 监听元素进入可视区域
 * 比传统的滚动事件监听更高效
 */
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      // 元素进入可视区域时的处理
      this.updateItem(entry.target);
    }
  });
}, {
  // 相对于哪个容器进行观察
  root: this.container,
  // 当元素10%进入可视区域时触发回调
  threshold: 0.1
});

// 需要观察的元素
const elementsToObserve = document.querySelectorAll('.scroll-item');
elementsToObserve.forEach(el => observer.observe(el));

3. 对象池技术 ​

javascript
/**
 * 对象池类
 * 用于复用对象，避免频繁创建和销毁，提高性能
 */
class ObjectPool {
  /**
   * 构造函数
   * @param {Function} createFn - 创建对象的函数
   * @param {Function} resetFn - 重置对象状态的函数
   */
  constructor(createFn, resetFn) {
    this.createFn = createFn;    // 创建新对象的函数
    this.resetFn = resetFn;      // 重置对象状态的函数
    this.pool = [];              // 对象池数组
  }
  
  /**
   * 从对象池中获取一个对象
   * 如果池为空，则创建新对象
   * @returns {*} - 对象
   */
  acquire() {
    if (this.pool.length === 0) {
      return this.createFn();
    }
    // 从池末尾取出对象（LIFO）
    return this.pool.pop();
  }
  
  /**
   * 将对象放回对象池
   * @param {*} obj - 要放回的对象
   */
  release(obj) {
    // 重置对象状态
    this.resetFn(obj);
    // 将对象放回池中
    this.pool.push(obj);
  }
  
  /**
   * 获取池中对象的数量
   * @returns {number} - 对象数量
   */
  size() {
    return this.pool.length;
  }
}

4. 估算行高 ​

对于高度不固定的项目，可以使用估算高度：

javascript
/**
 * 支持动态高度的虚拟滚动类
 */
class VirtualScrollDynamic {
  constructor(options) {
    // ... 其他初始化代码
    
    // 存储每个项目的实际高度
    this.heights = new Map();
    // 存储每个项目的累计高度（用于定位）
    this.offsets = new Map();
  }
  
  /**
   * 更新项目高度
   * 测量并缓存可见项目的实际高度
   */
  updateHeights() {
    const { startIndex, endIndex } = this.getVisibleRange();
    
    // 测量当前可见项目的实际高度
    for (let i = startIndex; i <= endIndex; i++) {
      // 如果还没有测量过这个项目的高度
      if (!this.heights.has(i)) {
        // 创建临时项目来测量高度
        const tempItem = this.renderItem(i);
        // 测量实际高度
        this.heights.set(i, tempItem.clientHeight);
        // 移除临时项目
        tempItem.remove();
      }
    }
    
    // 更新所有项目的偏移量
    this.updateOffsets();
    // 更新可见项目
    this.updateVisibleItems();
  }
  
  /**
   * 更新项目的累计偏移量
   * 根据每个项目的实际高度计算其位置
   */
  updateOffsets() {
    let currentOffset = 0;
    
    for (let i = 0; i < this.totalItems; i++) {
      // 设置项目的偏移量（距离顶部的距离）
      this.offsets.set(i, currentOffset);
      // 更新累计偏移量（加上当前项目的高度）
      currentOffset += this.heights.get(i) || 0;
    }
    
    // 更新滚动内容的总高度
    this.scrollContent.style.height = `${currentOffset}px`;
  }
  
  // ... 其他方法
}

最佳实践 ​

  1. 合理设置缓冲区：在可视区域外额外渲染少量元素，避免滚动时的白屏
  2. 避免频繁的 DOM 操作：使用文档片段批量更新
  3. 使用 transform 代替 top：transform 性能更好
  4. 优化渲染函数：避免在渲染函数中创建新对象
  5. 使用 memoization：缓存渲染结果，避免重复渲染

注意事项 ​

  1. 动态内容：内容变化时需要重新计算高度和位置
  2. 动画效果：虚拟滚动中的动画可能会影响性能
  3. 键盘导航：需要处理键盘事件与虚拟滚动的配合
  4. 响应式设计：确保在不同屏幕尺寸下都能正常工作

总结 ​

虚拟滚动是处理大量数据列表的必要技术。通过只渲染可视区域的元素，它可以显著提升性能，减少内存使用。无论是实现自定义虚拟滚动还是使用成熟的库，都需要理解其核心原理，并根据具体场景进行优化。

选择合适的虚拟滚动解决方案，可以让你的应用在处理大数据量时依然保持流畅的用户体验。
