大屏适配 ​

大屏适配（也称为数据可视化大屏适配）是指在各种尺寸的屏幕上，特别是大尺寸显示器（如电视墙、LED屏、监控屏等）上，让页面内容能够自适应显示并保持良好的视觉效果。

适配方案概述 ​

适配策略 ​

  1. 缩放适配（Scale）

    • 根据屏幕比例缩放整个容器
    • 保持内容的宽高比
    • 最常用的大屏适配方案

  2. CSS 弹性布局

    • 使用 rem/em 单位
    • 使用 viewport 单位
    • 使用 CSS Grid 和 Flexbox

  3. 媒体查询（Media Query）

    • 根据不同屏幕尺寸应用不同样式
    • 适合内容复杂度变化较大的场景

  4. 动态布局

    • 根据容器尺寸动态计算元素大小
    • 适合需要精确控制元素位置的场景

选择合适的方案 ​

场景推荐方案优点缺点
简单数据大屏缩放适配实现简单、效果好可能会模糊
复杂交互大屏动态布局精确控制、性能好实现复杂
多设备展示媒体查询灵活多样维护成本高
响应式设计弹性布局自适应性强需要精心设计

javascript
/**
 * 大屏适配工具类
 */
class ScreenAdapter {
  constructor(options = {}) {
    this.options = {
      designWidth: 1920,       // 设计稿宽度
      designHeight: 1080,      // 设计稿高度
      selector: '.screen-container', // 适配容器选择器
      mode: 'scale',           // 适配模式：scale/resize/custom
      resizeDelay: 100,       // 防抖延迟
      ...options
    };
    
    this.container = document.querySelector(this.options.selector);
    this.resizeTimer = null;
    
    this.init();
  }
  
  init() {
    if (!this.container) {
      console.error('未找到适配容器:', this.options.selector);
      return;
    }
    
    this.setupStyle();
    this.handleResize();
    this.bindEvents();
  }
  
  setupStyle() {
    // 设置基础样式
    this.container.style.position = 'absolute';
    this.container.style.left = '50%';
    this.container.style.top = '50%';
    this.container.style.transformOrigin = 'left top';
  }
  
  handleResize() {
    if (this.options.mode === 'scale') {
      this.scaleMode();
    } else if (this.options.mode === 'resize') {
      this.resizeMode();
    } else {
      this.customMode();
    }
  }
  
  scaleMode() {
    const screenWidth = window.innerWidth;
    const screenHeight = window.innerHeight;
    
    const widthScale = screenWidth / this.options.designWidth;
    const heightScale = screenHeight / this.options.designHeight;
    
    // 取较小的缩放比例，确保内容完全显示
    const scale = Math.min(widthScale, heightScale);
    
    // 应用缩放
    this.container.style.transform = `scale(${scale}) translate(-50%, -50%)`;
  }
  
  resizeMode() {
    const screenWidth = window.innerWidth;
    const screenHeight = window.innerHeight;
    
    const widthScale = screenWidth / this.options.designWidth;
    const heightScale = screenHeight / this.options.designHeight;
    
    // 调整容器尺寸
    const actualWidth = this.options.designWidth * Math.min(widthScale, heightScale);
    const actualHeight = this.options.designHeight * Math.min(widthScale, heightScale);
    
    this.container.style.width = `${actualWidth}px`;
    this.container.style.height = `${actualHeight}px`;
    this.container.style.transform = 'translate(-50%, -50%)';
    
    // 动态调整内部元素大小
    this.adjustElements(Math.min(widthScale, heightScale));
  }
  
  adjustElements(scale) {
    const elements = this.container.querySelectorAll('[data-scale]');
    elements.forEach(element => {
      const baseSize = parseFloat(element.dataset.scale);
      element.style.fontSize = `${baseSize * scale}px`;
    });
  }
  
  customMode() {
    // 自定义适配逻辑
    const screenWidth = window.innerWidth;
    const screenHeight = window.innerHeight;
    const ratio = screenWidth / screenHeight;
    const designRatio = this.options.designWidth / this.options.designHeight;
    
    // 根据屏幕比例调整布局
    if (ratio > designRatio) {
      // 屏幕更宽，以高度为基准
      const scale = screenHeight / this.options.designHeight;
      this.layoutByHeight(scale);
    } else {
      // 屏幕更高，以宽度为基准
      const scale = screenWidth / this.options.designWidth;
      this.layoutByWidth(scale);
    }
  }
  
  layoutByHeight(scale) {
    // 以高度为基准的布局逻辑
    this.container.style.height = '100vh';
    this.container.style.width = `${this.options.designWidth * scale}px`;
  }
  
  layoutByWidth(scale) {
    // 以宽度为基准的布局逻辑
    this.container.style.width = '100vw';
    this.container.style.height = `${this.options.designHeight * scale}px`;
  }
  
  bindEvents() {
    // 使用防抖优化性能
    const debouncedResize = () => {
      clearTimeout(this.resizeTimer);
      this.resizeTimer = setTimeout(() => {
        this.handleResize();
      }, this.options.resizeDelay);
    };
    
    window.addEventListener('resize', debouncedResize);
    
    // 监听设备方向变化
    window.addEventListener('orientationchange', debouncedResize);
    
    // 存储实例以便后续清除
    this.boundEvents = {
      resize: debouncedResize
    };
  }
  
  destroy() {
    // 清除事件监听
    if (this.boundEvents) {
      window.removeEventListener('resize', this.boundEvents.resize);
      window.removeEventListener('orientationchange', this.boundEvents.resize);
    }
    
    // 清除定时器
    if (this.resizeTimer) {
      clearTimeout(this.resizeTimer);
    }
  }
}

// 使用示例
const adapter = new ScreenAdapter({
  designWidth: 1920,
  designHeight: 1080,
  selector: '.screen-container',
  mode: 'scale'
});

/**
 * 基于 Rem 的适配方案

class RemAdapter {
  constructor(options = {}) {
    this.options = {
      baseWidth: 1920,      // 基准宽度
      baseFontSize: 16,     // 基础字体大小
      ...options
    };
    
    this.init();
  }
  
  init() {
    this.updateRem();
    this.bindEvents();
  }
  
  updateRem() {
    const screenWidth = window.innerWidth;
    // 计算当前缩放比例
    const scale = screenWidth / this.options.baseWidth;
    // 设置根元素字体大小
    const rootFontSize = this.options.baseFontSize * scale;
    document.documentElement.style.fontSize = `${rootFontSize}px`;
  }
  
  bindEvents() {
    window.addEventListener('resize', () => {
      this.updateRem();
    });
  }
}

import React, { useEffect, useRef, useState } from 'react';

/**
 * 大屏适配组件
 */
const ScreenAdapter = ({ 
  children, 
  designWidth = 1920, 
  designHeight = 1080,
  mode = 'scale',
  className = '' 
}) => {
  const containerRef = useRef(null);
  const [scale, setScale] = useState(1);
  
  useEffect(() => {
    const handleResize = () => {
      const screenWidth = window.innerWidth;
      const screenHeight = window.innerHeight;
      
      if (mode === 'scale') {
        const widthScale = screenWidth / designWidth;
        const heightScale = screenHeight / designHeight;
        const newScale = Math.min(widthScale, heightScale);
        setScale(newScale);
      }
    };
    
    // 初始化
    handleResize();
    
    // 添加事件监听
    window.addEventListener('resize', handleResize);
    
    // 清理函数
    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, [designWidth, designHeight, mode]);
  
  const containerStyle = {
    transform: mode === 'scale' 
      ? `scale(${scale}) translate(-50%, -50%)` 
      : undefined,
    position: 'absolute',
    left: '50%',
    top: '50%',
    transformOrigin: mode === 'scale' ? 'left top' : undefined,
    width: mode === 'resize' ? `${designWidth * scale}px` : undefined,
    height: mode === 'resize' ? `${designHeight * scale}px` : undefined,
  };
  
  return (
    <div 
      ref={containerRef}
      className={`screen-container ${className}`}
      style={containerStyle}
    >
      {children}
    </div>
  );
};

// 自定义 Hook
export const useScreenAdapter = (options = {}) => {
  const [screenSize, setScreenSize] = useState({
    width: window.innerWidth,
    height: window.innerHeight,
    scale: 1
  });
  
  useEffect(() => {
    const handleResize = () => {
      const width = window.innerWidth;
      const height = window.innerHeight;
      
      const designWidth = options.designWidth || 1920;
      const designHeight = options.designHeight || 1080;
      
      const widthScale = width / designWidth;
      const heightScale = height / designHeight;
      const scale = Math.min(widthScale, heightScale);
      
      setScreenSize({
        width,
        height,
        scale
      });
    };
    
    handleResize();
    window.addEventListener('resize', handleResize);
    
    return () => window.removeEventListener('resize', handleResize);
  }, [options]);
  
  return screenSize;
};

// 使用示例
const Dashboard = () => {
  const { scale } = useScreenAdapter({
    designWidth: 1920,
    designHeight: 1080
  });
  
  return (
    <div style={{ 
      background: '#000', 
      width: '100vw', 
      height: '100vh',
      overflow: 'hidden'
    }}>
      <ScreenAdapter designWidth={1920} designHeight={1080}>
        <Header />
        <Content scale={scale} />
      </ScreenAdapter>
    </div>
  );
};

// 响应式组件
const ResponsiveComponent = ({ children }) => {
  const { scale } = useScreenAdapter();
  
  return (
    <div 
      style={{ 
        transform: `scale(${scale})`,
        transformOrigin: 'top left'
      }}

};

export default {
  name: 'ScreenAdapter',
  props: {
    designWidth: {
      type: Number,
      default: 1920
    },
    designHeight: {
      type: Number,
      default: 1080
    },
    mode: {
      type: String,
      default: 'scale',
      validator: (value) => ['scale', 'resize'].includes(value)
    }
  },
  data() {
    return {
      scale: 1,
      screenWidth: window.innerWidth,
      screenHeight: window.innerHeight
    }
  },
  computed: {
    containerStyle() {
      if (this.mode === 'scale') {
        return {
          transform: `scale(${this.scale}) translate(-50%, -50%)`,
          position: 'absolute',
          left: '50%',
          top: '50%',
          transformOrigin: 'left top'
        }
      } else {
        return {
          position: 'absolute',
          left: '50%',
          top: '50%',
          width: `${this.designWidth * this.scale}px`,
          height: `${this.designHeight * this.scale}px`,
          transform: 'translate(-50%, -50%)'
        }
      }
    }
  },
  mounted() {
    this.updateScale();
    window.addEventListener('resize', this.handleResize);
  },
  beforeDestroy() {
    window.removeEventListener('resize', this.handleResize);
  },
  methods: {
    updateScale() {
      this.screenWidth = window.innerWidth;
      this.screenHeight = window.innerHeight;
      
      const widthScale = this.screenWidth / this.designWidth;
      const heightScale = this.screenHeight / this.designHeight;
      
      this.scale = Math.min(widthScale, heightScale);
    },
    handleResize() {
      // 使用防抖优化
      clearTimeout(this.resizeTimer);
      this.resizeTimer = setTimeout(() => {
        this.updateScale();
      }, 100);
    }
  }
}
</script>

      transform: `scale(${scale.value}) translate(-50%, -50%)`,
      position: 'absolute',
      left: '50%',

      transformOrigin: 'left top'

    return {
      position: 'absolute',
      left: '50%',
      top: '50%',
      width: `${props.designWidth * scale.value}px`,
      height: `${props.designHeight * scale.value}px`,
      transform: 'translate(-50%, -50%)'
    }
  }
})

  screenWidth.value = window.innerWidth
  screenHeight.value = window.innerHeight

}

onMounted(() => {

})

onBeforeUnmount(() => {
  window.removeEventListener('resize', handleResize)
  clearTimeout(window.resizeTimer)

// 暴露方法供父组件调用
defineExpose({

class FontSizeAdapter {

  }
}

class GridLayoutAdapter {
  constructor(container, options = {}) {
    this.container = container;

    const gap = this.options.gap;

    const columnWidth = (containerWidth - totalGap) / this.options.columns;
    
    // 更新容器的Grid样式

/**

class MultiScreenAdapter {
  constructor() {
    this.screens = new Map();
    this.currentScreen = 'default';
    this.init();
  }
  
  init() {
    this.detectScreen();
    window.addEventListener('resize', () => {
      clearTimeout(this.resizeTimer);
      this.resizeTimer = setTimeout(() => {
        this.detectScreen();
      }, 300);
    });
  }
  
  detectScreen() {
    const width = window.innerWidth;
    const height = window.innerHeight;
    
    // 屏幕尺寸分类
    const screenTypes = {
      desktop: { min: 1920, max: Infinity },
      laptop: { min: 1366, max: 1919 },
      tablet: { min: 768, max: 1365 },
      mobile: { min: 0, max: 767 }
    };
    
    let newScreenType = 'default';
    for (const [type, range] of Object.entries(screenTypes)) {
      if (width >= range.min && width <= range.max) {
        newScreenType = type;
        break;
      }
    }
    
    if (newScreenType !== this.currentScreen) {
      this.switchScreen(newScreenType);
    }
  }
  
  switchScreen(type) {
    this.currentScreen = type;
    
    // 移除旧的样式
    const oldStyles = document.querySelectorAll('.screen-style');
    oldStyles.forEach(style => style.remove());
    
    // 添加新的样式
    const style = document.createElement('style');
    style.className = 'screen-style';
    style.textContent = this.getScreenStyles(type);
    document.head.appendChild(style);
    
    // 触发自定义事件
    const event = new CustomEvent('screenChange', {
      detail: { type: this.currentScreen }
    });
    window.dispatchEvent(event);
  }
  
  getScreenStyles(type) {
    const styles = {
      desktop: `
        .header { font-size: 24px; }
        .chart { width: 100%; height: 400px; }
      `,
      laptop: `
        .header { font-size: 20px; }
        .chart { width: 100%; height: 350px; }
      `,
      tablet: `
        .header { font-size: 18px; }
        .chart { width: 100%; height: 300px; }
        .grid-container { grid-template-columns: repeat(6, 1fr); }
      `,
      mobile: `
        .header { font-size: 16px; }
        .chart { width: 100%; height: 250px; }
        .grid-container { grid-template-columns: repeat(3, 1fr); }
      `
    };
    
    return styles[type] || styles.desktop;
  }
}

// 使用防抖
function debounce(func, wait) {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}

// 使用 RAF (RequestAnimationFrame)
function throttle(func) {
  let running = false;
  return function(...args) {
    if (running) return;
    running = true;
    requestAnimationFrame(() => {
      func.apply(this, args);
      running = false;
    });
  };
}

        // 自定义匹配逻辑
        return width / height > 1.6;
    },
    adapt: function(element, options) {
        const { baseWidth, baseHeight } = options;
        const ratio = window.innerWidth / window.innerHeight;
        
        if (ratio > baseWidth / baseHeight) {
            // 屏幕更宽
            element.style.width = '100vw';
            element.style.height = `${baseHeight * window.innerWidth / baseWidth}px`;
        } else {
            // 屏幕更高
            element.style.height = '100vh';
            element.style.width = `${baseWidth * window.innerHeight / baseHeight}px`;
        }
    }
};

    })
  }
